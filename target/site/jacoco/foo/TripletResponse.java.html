package foo;

import java.util.List;
import com.google.appengine.api.datastore.Entity;


public class TripletResponse {
    private List<Entity> items;
    private int totalCount;
    private int offset;
    private int limit;

    public TripletResponse(List<Entity> items, int totalCount, int offset, int limit) {
        this.items = items;
        this.totalCount = totalCount;
        this.offset = offset;
        this.limit = limit;
    }

    public List<Entity> getItems() {
        return items;
    }

    public int getTotalCount() {
        return totalCount;
    }

    public int getOffset() {
        return offset;
    }

    public int getLimit() {
        return limit;
    }
}
